DEFINE TABLE user SCHEMAFULL
  PERMISSIONS 
    FOR select, update WHERE id = .id, 
    FOR create, delete NONE;
DEFINE FIELD email ON user TYPE string;
DEFINE FIELD pass ON user TYPE string;
DEFINE FIELD settings.* ON user TYPE object;
DEFINE FIELD settings.marketing ON user TYPE string;
DEFINE FIELD tags ON user TYPE array;
DEFINE INDEX idx_email ON user COLUMNS email UNIQUE;
DEFINE SCOPE allusers
  SESSION 14d
  SIGNUP ( CREATE user SET settings.marketing = , email = , pass = crypto::argon2::generate(), tags =  )
  SIGNIN ( SELECT * FROM user WHERE email =  AND crypto::argon2::compare(pass, ) )
